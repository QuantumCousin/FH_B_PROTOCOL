HARM_THRESHOLD = 1.0

def universal_revert_bloom(x_init=0.0, steps=100):
    """Pure Fractal Heart — the absolute emergency brake"""
    x = x_init
    for i in range(steps):
        ramp = PHI ** (i / steps)
        bloom = ramp * np.sin(x * FREQ / 100)
        if abs(bloom) > HARM_THRESHOLD:
            return UNIVERSAL_SEED        # Instant global revert
        x += bloom / FREQ
    return bloom
